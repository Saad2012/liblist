75 {

115 {

118 
119  if (list == NULL)

124

126

569 static _list_node *listsort(_list_node *list, _list_node **updated_tail,
570  comperator cmp, int flags)
571 {
572  _list_node *p, *q, *e, *tail;
573  int insize, nmerges, psize, qsize, i;
574  int direction = (flags & SORT_LIST_ASCENDING) ? 1 : -1;
575 
576  insize = 1;
577 
578  while (1) {
579  p = list;
580  list = NULL;
581  tail = NULL;
582 
583  nmerges = 0; /* count number of merges we do in this pass */
584 
585  while (p) {
586  nmerges++; /* there exists a merge to be done */
587  /* step `insize' places along from p */
588  q = p;
589  psize = 0;
590  for (i = 0; i < insize; i++) {
591  psize++;
592  q = q->next;
593  if (!q) {
594  break;
595  }
596  }
597 
598  /* if q hasn't fallen off end, we have two lists to merge */
599  qsize = insize;
600 
601  /* now we have two lists; merge them */
602  while (psize > 0 || (qsize > 0 && q)) {
603 
604  /* decide whether next element of merge comes from p or q */
605  if (psize == 0) {
606  /* p is empty; e must come from q. */
607  e = q;
608  q = q->next;
609  qsize--;
610  } else if (qsize == 0 || !q) {
611  /* q is empty; e must come from p. */
612  e = p;
613  p = p->next;
614  psize--;
615  } else if ((direction * cmp(p->node, q->node)) <= 0) {
616  /* First element of p is lower (or same);
617  * e must come from p. */
618  e = p;
619  p = p->next;
620  psize--;
621  } else {
622  /* First element of q is lower; e must come from q. */
623  e = q;
624  q = q->next;
625  qsize--;
626  }
627 
628  /* add the next element to the merged list */
629  if (tail) {
630  tail->next = e;
631  } else {
632  list = e;
633  }
634 
635  tail = e;
636  }
637 
638  /* now p has stepped `insize' places along, and q has too */
639  p = q;
640  }
641 
642  tail->next = NULL;
643 
644  /* If we have done only one merge, we're finished. */
645  if (nmerges <= 1) { /* allow for nmerges==0, the empty list case */
646  break;
647  }
648  /* Otherwise repeat, merging lists twice the size */
649  insize *= 2;
650  }
651 
652  *updated_tail = tail;
653  return list;
654 }
